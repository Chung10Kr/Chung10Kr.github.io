```javascript

let arr = [
    [0,0,0,0,1],
    [0,0,0,0,3],
    [0,0,0,0,4],
    [0,2,0,0,2],
    [4,5,0,2,0]
]
// 이거를 왼쪽으로 90도 회전 해서

/*
[
  [ 1, 3, 4, 2, 0 ],
  [ 0, 0, 0, 0, 2 ],
  [ 0, 0, 0, 0, 0 ],
  [ 0, 0, 0, 2, 5 ],
  [ 0, 0, 0, 0, 4 ]
] 이렇게
*/
let result = [];
for( var i=0; i< arr.length; i++ ){
    result[i] = new Array( arr[0].length )
};  //빈 arr 생성

/*
0,4 -> 0,0
1,4 -> 0,1
2,4 -> 0,2
3,4 -> 0,3
4,4 -> 0,4

0,3 -> 1,0
1,3 -> 1,1
2,3 -> 1,2
3,3 -> 1,3
4,3 -> 1,4

0,2 -> 2,0
1,2 -> 2,1
2,2 -> 2,2
3,2 -> 2,3
4,2 -> 2,4

*/

for( var i=0; i< arr.length ; i++ )
{
    for( var j=0; j< arr[0].length ; j++ )
    {
        result[i][j] = arr[j][ arr[0].length-1-i ]
    }
}

console.log( result )
```
